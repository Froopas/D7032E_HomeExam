package Language;

import java.util.ArrayList;

import Util.Dimension;

public class LanguageInfo {

    private ArrayList<String> dictionaries;
    private ArrayList<Dimension> dimensions;
    
    public LanguageInfo(ArrayList<String> dictionaries, ArrayList<Dimension> dimensions) {
        this.dictionaries = dictionaries;
        this.dimensions = dimensions;
    }

    public ArrayList<Dimension> getDimensions() {
        return dimensions;
    }

    public ArrayList<String> getDictionaries() {
        return dictionaries;
    }

    public void setDictionaries(ArrayList<String> dictionaries) {
        this.dictionaries = dictionaries;
    }

    public void setDimensions(ArrayList<Dimension> dimensions) {
        this.dimensions = dimensions;
    }
}
